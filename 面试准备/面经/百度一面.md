function num(arr) {
  var newArr = Array.from({length: 10}).fill(0);
  let big = 0, small = 0; //整数部分，和个位数
  arr.forEach(item => {
    if(item < 10) {
      newArr[item]++;
    } else {
      small = item % 10;
      big = (item - small) / 10;
      newArr[small]++;
      newArr[big]++;
    }
  })
  return newArr;
}
